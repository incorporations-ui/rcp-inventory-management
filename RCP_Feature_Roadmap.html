SQL Quick Reference — Supabase
Useful queries to run in the SQL Editor for setup, debugging, and resets
1. Create first admin user (run after creating user in Auth)
-- Replace the UUID and name with your actual values from Auth → Users
INSERT INTO user_profiles (id, full_name, role, is_active)
VALUES (
  'paste-your-auth-user-uuid-here',
  'Your Full Name',
  'admin',
  true
);
2. Check current stock summary
SELECT sku_code, display_name, total_units, so_reserved_units,
       manual_reserved_units, truly_available_units
FROM stock_summary
ORDER BY truly_available_units ASC;  -- lowest available first
3. Check audit log — last 50 changes
SELECT al.table_name, al.action, al.changed_at,
       up.full_name AS changed_by,
       al.old_data, al.new_data
FROM audit_log al
LEFT JOIN user_profiles up ON up.id = al.changed_by
ORDER BY al.changed_at DESC
LIMIT 50;
4. Reset ALL transaction data (keep masters) — USE WITH CAUTION
-- This wipes all POs, GRNs, SOs, invoices, stock, lots, movements etc.
-- Masters (SKUs, brands, racks, customers, suppliers) are preserved.
-- Sequences are reset to 0.
SELECT reset_transaction_data();
5. Fix stock master if numbers seem wrong
-- Recalculate total_units from stock_movements ledger
UPDATE stock_master sm
SET total_units = (
  SELECT COALESCE(SUM(units_in) - SUM(units_out), 0)
  FROM stock_movements
  WHERE sku_id = sm.sku_id
),
last_updated = NOW();

-- Then check results
SELECT * FROM stock_summary WHERE truly_available_units < 0;
6. View all active reservations with customer names
SELECT r.reservation_no, c.name AS customer, s.display_name AS sku,
       r.reserved_units, r.reserved_until, r.status, up.full_name AS reserved_by
FROM manual_reservations r
JOIN customers c  ON c.id = r.customer_id
JOIN skus s       ON s.id = r.sku_id
JOIN user_profiles up ON up.id = r.reserved_by
WHERE r.status = 'active'
ORDER BY r.created_at;
7. Stock ageing summary (quick view)
SELECT age_bucket,
       COUNT(*) AS lots,
       SUM(remaining_units) AS total_units,
       ROUND(SUM(stock_value)::numeric, 2) AS total_value
FROM stock_ageing
GROUP BY age_bucket
ORDER BY MIN(age_days);
8. Find all GRN lines that are still pending (shouldn't be any after finalization)
SELECT g.grn_number, gl.sku_id, s.display_name, gl.expected_units, gl.status
FROM grn_lines gl
JOIN grns g     ON g.id = gl.grn_id
JOIN skus s     ON s.id = gl.sku_id
WHERE gl.status = 'pending'
AND   g.status  = 'finalized';  -- these are data inconsistencies
9. Document sequence current state
SELECT doc_type, prefix, current_number, fiscal_year,
       prefix || '-' || fiscal_year || '-' || LPAD((current_number+1)::text, 5, '0') AS next_number
FROM document_sequences
ORDER BY doc_type;
